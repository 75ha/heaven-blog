用正则将HTML转为BBCode

Nov 6, 2014

为了防止富文本编辑器中用户的输入被不轨的人用于XSS攻击，一般的做法是只接受某种特定语法，比如markdown、BBCode。一般网站运营初始阶段对此考虑都不太周全，都是在网站发展到一定规模以后才会考虑到这个防XSS这个问题，所以就有将以前用户输入的html代码都转换为BBCode这一类的需求。接下来我将介绍一下，如何将已有的html代码通过正则表达式转换为BBCode。

//在block元素前的inline元素之后，填入一个换行符
$str = preg_replace('/(<\/(strong|u|em|a|span)>|<img([^<]*?)>)([^<]*?)<(h\d|p|div|table|pre)(?:[^<]*?)>/i',
    "$1$4\r\n<$5>", $str);

$patterns = array(
    '/<!--(.*?)-->/i',
    '/<strong(?:[^<]*?)>([^<]*?)<\/strong>/i',
    '/<u(?:[^<]*?)>([^<]*?)<\/u>/i',
    '/<em(?:[^<]*?)>([^<]*?)<\/em>/i',
    '/<br\s*\/?>/i',
    '/<img (?:[^<]*?)src="([^<]*?)"(?:[^<]*?)\/>/i',
    '/<a(?:[^<]*?)href="([^<]*?)"(?:[^<]*?)>([^<]*?)<\/a>/i',
    '/<span style="font-size:([^<]*?)">([^<]*?)<\/span>/i',
    '/<span style="color:([^<]*?)">([^<]*?)<\/span>/i',
    '/<span style="background-color:([^<]*?)">([^<]*?)<\/span>/i',
    '/<span style="line-height:([^<]*?)">([^<]*?)<\/span>/i',
    '/<div(?:[^<]*?)>([^<]*?)<\/div>/i',
    '/<(p|table|td|tr|thead|tbody|s|pre)(?:[^<]*?)>([^<]*?)<\/\1>/i',
    '/<(h\d)(?:[^<]*?)>([^<]*?)<\/\1>/i',
    '/<(script)([^<]*?)>([^<]*?)<\/\1>/i',
    '/<embed([^<]*?)\/?>/i',
    '/<ol(?:[^<]*?)>([^<]*?)<\/ol>/i',
    '/<ul(?:[^<]*?)>([^<]*?)<\/ul>/i',
    '/<li(?:[^<]*?)>([^<]*?)<\/li>/i',
    '/<font color="([^<]*?)"(?:[^<]*?)>([^<]*?)<\/font>/i'
);
$replacements = array(
    '',
    '[b]$1[/b]',
    '[u]$1[/u]',
    '[i]$1[/i]',
    "\r\n",
    '[img]$1[/img]',
    '[url=$1]$2[/url]',
    '[size=$1]$2[/size]',
    '[color=$1]$2[/color]',
    '$2',
    '$2',
    "$1\r\n",
    "$2",
    "[b][size=22px]$2[/size][/b]\r\n",
    '',
    '[list=1]$1[/list]',
    '[list]$1[/list]',
    '[*]$1',
    '$1',
);
for ($i = 0; $i < 10; $i++) {
    $str = preg_replace($patterns, $replacements, $str);
}

//处理比较麻烦的size,color属性，他俩可能会参杂其他属性
$patterns = array(
    '/\[color=([^\[]*?); font-size:([^\[]*?)\]([^\[]*?)\[\/color\]/i',
    '/\[size=([^\[]*?); color:([^\[]*?)\]([^\[]*?)\[\/size\]/i',
);
$replacements = array(
    '[size=$2][color=$1]$3[/color][/size]',
    '[color=$2][size=$1]$3[/size][/color]',
);
$str = preg_replace($patterns, $replacements, $str);
